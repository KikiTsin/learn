如何将数组重排 快速排序（前序遍历）
```javascript
// 三路切分 
function getSortedList (arr) {
    function swap (j, k) {
        let temp = arr[k]
        arr[k] = arr[j]
        arr[j] = temp
    }
    // b-->begin   e--->end
    function toSort (arr, b, e) {
        if (b >= e || (b + 1) >= e) {
            return
        }
        let mid = b + ((e-b)>>1)
        let x = arr[mid]
        let l = b, i = b, r = e - 1
        while (i <= r) {
            if (arr[i] < x) {
                swap(l++, i++)
            } else if (arr[i] === x) {
                i++
            } else {
                swap(r--, i)
            }
        }
        toSort(arr, b, l)
        toSort(arr, i, e)
        return arr
    }
    toSort(arr, 0, arr.length)
    return arr
}
getSortedList([1,5,2,5,7,8])
```
